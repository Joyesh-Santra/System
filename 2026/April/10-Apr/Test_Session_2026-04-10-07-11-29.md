## Construct BST from Preorder Traversal

Given a preorder traversal of a BST, construct the tree and output its **inorder traversal**.

**Input:** Line 1: `n`. Line 2: `n` integers (preorder).

**Output:** Inorder traversal (should be sorted).

**Constraints:** `1 в‰¤ n в‰¤ 100`, `0 в‰¤ val в‰¤ 10^8`, all values distinct.

#### рџ’» My Solution

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

struct Node { int val; Node *left, *right; Node(int v): val(v), left(nullptr), right(nullptr){}; };

Node* insert(Node* root, int val) {
    if(!root) return new Node(val);
    if(val < root->val) root->left = insert(root->left, val);
    else root->right = insert(root->right, val);
    return root;
}

void inorder(Node* root, vector<int>& res) {
    if(!root) return;
    inorder(root->left, res);
    res.push_back(root->val);
    inorder(root->right, res);
}

int main() {
    int n; cin >> n;
    Node* root = nullptr;
    for(int i=0;i<n;i++){int x;cin>>x;root=insert(root,x);}
    vector<int> res; inorder(root, res);
    for(int i=0;i<(int)res.size();i++) cout << res[i] << (i+1<(int)res.size()?' ':'\n');
    return 0;
```
